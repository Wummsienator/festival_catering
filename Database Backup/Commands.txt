#Create Backup

BACKUP DATABASE FestivalCateringAPP
TO DISK = 'C:\SQLBackups\FestivalCateringAPP_Initial.bak'
WITH INIT;

#Load Backup
ALTER DATABASE FestivalCateringAPP SET SINGLE_USER WITH ROLLBACK IMMEDIATE;

RESTORE DATABASE FestivalCateringAPP
FROM DISK = 'C:\SQLBackups\FestivalCateringAPP_Initial.bak'
WITH REPLACE;

ALTER DATABASE FestivalCateringAPP SET MULTI_USER;

#Check if database is used
SELECT 
    s.session_id,
    s.login_name,
    s.host_name,
    s.program_name
FROM sys.dm_exec_sessions s
WHERE s.database_id = DB_ID('FestivalCateringAPP');

#Kill sessions
USE master;
GO
KILL 77;
GO
